<script type="text/javascript">
  
  function calcProductAvg(){   
     
         let qty_wdtTotal = 0 ;
         let pcs_wdtTotal = 0 ;
    	 let bundle_pack_wdtTotal = 0 ;
    
         document.querySelectorAll('.allproduct_calc').forEach(function(element, index, array){
           let wdt_org_qty = 0;
           
           let bundle_pack = 0;
           
           let allproduct = document.querySelectorAll('.allproduct_calc')[index];
           
           let qty_val = allproduct.querySelector(".product_item_qty").value;
           
           console.log("INPUT-VALUE: ", qty_val)
           
           
           
           if(qty_val > 0){
             let trtd = allproduct.querySelector('td[data-quantity]');
           	 let price_val = allproduct.querySelector('.td_text-price').getAttribute('data-price');
             
             let lengthtd = allproduct.querySelectorAll(".input_item").length;
             if(lengthtd > 0){
               for (let i = 0; i < lengthtd; i++) {
                 let dataQuantity = allproduct.querySelectorAll(".input_item")[i].getAttribute('data-quantity');
                 
                 wdt_org_qty += parseInt(dataQuantity);
                 
               }
               
             } 
             
              for (let i = 0; i <= lengthtd; i++) {                
                 bundle_pack += parseInt(qty_val);                 
              }
             
             let qtyOp = wdt_org_qty*qty_val;
             
             pcs_wdtTotal += qtyOp;
             

             let listPrice = (parseFloat(qtyOp) * parseFloat(price_val));
             allproduct.querySelector('.pack__qty').innerHTML = wdt_org_qty*qty_val;
             allproduct.querySelector('.pack_amount').innerHTML = listPrice.toFixed(2);
             
             qty_wdtTotal = (parseFloat(qty_wdtTotal) + parseFloat(listPrice));
             
             allproduct.querySelector(".product_item_qty").setAttribute("data-allqty", (wdt_org_qty*qty_val));
             allproduct.querySelector(".product_item_qty").setAttribute("data-price", listPrice.toFixed(2));
             
           }
           
           bundle_pack_wdtTotal = bundle_pack;
         });
    	
    	 document.getElementById("total_amount").innerHTML = qty_wdtTotal.toFixed(2);
         document.getElementById("total_qty").innerHTML = pcs_wdtTotal; 
         document.getElementById("total_pack").innerHTML = bundle_pack_wdtTotal;
    
  }
  
  const divs = document.querySelectorAll('.product_item_qty');
  divs.forEach(el => el.addEventListener('change', event => {
      document.getElementById("all_input_qty").value = '0';
      calcProductAvg();
  }));
 
  document.getElementById('all_input_qty').addEventListener('change', function(event) {
      var qtyp = document.getElementById("all_input_qty").value;
      let lengthI = document.querySelectorAll(".product_item_qty").length;
      if(lengthI > 0){
         for (let i = 0; i < lengthI; i++) {
           document.querySelectorAll(".product_item_qty")[i].value = qtyp;
         }
      }
      calcProductAvg();
  })
   
  
  /* ADD TO CART FUNCTION */
  
  function addTocartScript(formData){
    setTimeout( function() {
          var url =  '/cart/add.js';
          fetch(url, {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json'
            },
            body: JSON.stringify(formData)
          })
          .then(response => {
            window.location.href="/cart";
            return response.json();
          })
          .catch((error) => {
            console.error('Error:', error);
          }); 
        }, 300 );
  }

  const add_to_cart =  document.getElementById("button_cart"); 
  add_to_cart.addEventListener("click", function(event){
    var productItem = [];
    
    document.querySelectorAll('.product_item_qty[value]').forEach(function(element, index, array){
      var qty = element.value;
      if(qty > 0){
        let variantId = element.getAttribute('data-variantId');
        let allqty = element.getAttribute('data-allqty');
        let price = element.getAttribute('data-price');
        let title = element.getAttribute('data-title');        
        productItem.push({"id": variantId, "quantity": allqty, "properties": {"TITLE": title, "AMOUNT": price,"PACK": qty}});
      }      
    });    
    var formData = {'items': productItem};      
    addTocartScript(formData);    
    
  });

</script>
