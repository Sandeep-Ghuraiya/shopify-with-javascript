                	if (!parent.classList.contains('open-toggle')) {
                		parent.classList.add('open-toggle');
                		article.style.maxHeight = article.scrollHeight + 'px';

                		if (this === accordion_toggle[i] && !parent.classList.contains('open-toggle')) {
